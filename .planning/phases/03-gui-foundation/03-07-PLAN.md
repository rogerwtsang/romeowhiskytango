```python
"""Dark triadic color theme for the baseball simulator GUI.

Triadic scheme using:
- Primary: Deep navy blue (#05285b)
- Secondary: Light blue (#81b4fe)
- Accent: Amber/gold (#ffc107)
- Text: Light lavender (#cbbbe9)
"""

import tkinter as tk
from tkinter import ttk

# Color palette (from mockup)
COLORS = {
    'bg_dark': '#1a1a2e',        # Darkest background
    'bg_medium': '#16213e',      # Medium background (panels)
    'bg_light': '#0f3460',       # Lighter background (inputs)
    'primary': '#05285b',        # Primary navy
    'secondary': '#81b4fe',      # Light blue
    'accent': '#ffc107',         # Amber/gold
    'text_primary': '#e8e8e8',   # Light gray text
    'text_secondary': '#cbbbe9', # Lavender text
    'text_muted': '#8888aa',     # Muted text
    'border': '#2a2a4a',         # Border color
    'success': '#4caf50',        # Green for success
    'error': '#f44336',          # Red for errors
    'button_bg': '#05285b',      # Button background
    'button_fg': '#cbbbe9',      # Button text
    'button_active': '#0a3d8f',  # Button hover
}


def apply_dark_triadic_theme(root: tk.Tk) -> ttk.Style:
    """Apply dark triadic theme to the application.

    Args:
        root: The root Tk window

    Returns:
        Configured ttk.Style instance
    """
    style = ttk.Style(root)

    # Configure root window
    root.configure(bg=COLORS['bg_dark'])

    # Try to use clam theme as base (better for customization)
    try:
        style.theme_use('clam')
    except tk.TclError:
        pass

    # Frame styles
    style.configure('TFrame', background=COLORS['bg_dark'])
    style.configure('Panel.TFrame', background=COLORS['bg_medium'])

    # Label styles
    style.configure('TLabel',
                    background=COLORS['bg_dark'],
                    foreground=COLORS['text_primary'])
    style.configure('Header.TLabel',
                    background=COLORS['bg_dark'],
                    foreground=COLORS['text_secondary'],
                    font=('TkDefaultFont', 12, 'bold'))
    style.configure('Muted.TLabel',
                    background=COLORS['bg_dark'],
                    foreground=COLORS['text_muted'])

    # LabelFrame styles
    style.configure('TLabelframe',
                    background=COLORS['bg_medium'],
                    foreground=COLORS['text_secondary'])
    style.configure('TLabelframe.Label',
                    background=COLORS['bg_medium'],
                    foreground=COLORS['accent'])

    # Button styles
    style.configure('TButton',
                    background=COLORS['button_bg'],
                    foreground=COLORS['button_fg'],
                    borderwidth=1,
                    relief='flat',
                    padding=(10, 5))
    style.map('TButton',
              background=[('active', COLORS['button_active']),
                         ('pressed', COLORS['primary'])],
              foreground=[('active', COLORS['text_primary'])])

    # Accent button (Run Simulation)
    style.configure('Accent.TButton',
                    background=COLORS['accent'],
                    foreground=COLORS['bg_dark'],
                    font=('TkDefaultFont', 10, 'bold'))
    style.map('Accent.TButton',
              background=[('active', '#e5ac00'),
                         ('pressed', '#cc9900')])

    # Entry styles
    style.configure('TEntry',
                    fieldbackground=COLORS['bg_light'],
                    foreground=COLORS['text_primary'],
                    insertcolor=COLORS['text_primary'])

    # Combobox styles
    style.configure('TCombobox',
                    fieldbackground=COLORS['bg_light'],
                    foreground=COLORS['text_primary'],
                    background=COLORS['bg_medium'],
                    arrowcolor=COLORS['accent'])

    # Notebook (tabs) styles
    style.configure('TNotebook',
                    background=COLORS['bg_dark'],
                    borderwidth=0)
    style.configure('TNotebook.Tab',
                    background=COLORS['bg_medium'],
                    foreground=COLORS['text_secondary'],
                    padding=(15, 8))
    style.map('TNotebook.Tab',
              background=[('selected', COLORS['primary'])],
              foreground=[('selected', COLORS['text_primary'])])

    # Progressbar styles
    style.configure('TProgressbar',
                    background=COLORS['accent'],
                    troughcolor=COLORS['bg_light'])

    # Scrollbar styles
    style.configure('TScrollbar',
                    background=COLORS['bg_medium'],
                    troughcolor=COLORS['bg_dark'],
                    arrowcolor=COLORS['text_muted'])

    # Spinbox styles
    style.configure('TSpinbox',
                    fieldbackground=COLORS['bg_light'],
                    foreground=COLORS['text_primary'],
                    arrowcolor=COLORS['accent'])

    # Checkbutton styles
    style.configure('TCheckbutton',
                    background=COLORS['bg_dark'],
                    foreground=COLORS['text_primary'])

    # Scale styles (for sliders)
    style.configure('TScale',
                    background=COLORS['bg_dark'],
                    troughcolor=COLORS['bg_light'])

    # PanedWindow styles
    style.configure('TPanedwindow',
                    background=COLORS['border'])

    # Separator styles
    style.configure('TSeparator',
                    background=COLORS['border'])

    return style
```

```python
"""Tabbed simulation panel containing Lineup and Visuals views."""

import tkinter as tk
from tkinter import ttk
from typing import Optional, Callable, List, Dict, Any
from src.models.player import Player
from src.gui.widgets.lineup_builder import LineupBuilder


class SimulationPanel(ttk.Frame):
    """Tabbed panel for simulation configuration and visualization.

    Contains:
    - Header with title, tab selectors, and Run button
    - Lineup tab: Lineup builder and controls
    - Visuals tab: Charts and visualizations (placeholder for now)
    """

    def __init__(
        self,
        parent,
        on_run: Optional[Callable[[], None]] = None,
        on_compare: Optional[Callable[[], None]] = None,
        **kwargs
    ):
        """Initialize SimulationPanel.

        Args:
            parent: Parent widget
            on_run: Callback for Run Simulation button
            on_compare: Callback for Compare Mode toggle
        """
        super().__init__(parent, **kwargs)

        self.on_run = on_run
        self.on_compare = on_compare

        # Configure grid
        self.columnconfigure(0, weight=1)
        self.rowconfigure(0, weight=0)  # Header
        self.rowconfigure(1, weight=1)  # Tab content

        self._create_header()
        self._create_tabs()

    def _create_header(self):
        """Create header with title, tabs, and Run button."""
        header = ttk.Frame(self)
        header.grid(row=0, column=0, sticky='ew', padx=5, pady=5)
        header.columnconfigure(1, weight=1)  # Spacer expands

        # Title
        ttk.Label(
            header,
            text="Simulation",
            style='Header.TLabel'
        ).grid(row=0, column=0, sticky='w', padx=(5, 20))

        # Tab selector buttons (custom tab-like buttons)
        tab_frame = ttk.Frame(header)
        tab_frame.grid(row=0, column=1, sticky='w')

        self.lineup_tab_btn = ttk.Button(
            tab_frame,
            text="Lineup",
            command=lambda: self._switch_tab('lineup'),
            width=10
        )
        self.lineup_tab_btn.pack(side=tk.LEFT, padx=2)

        self.visuals_tab_btn = ttk.Button(
            tab_frame,
            text="Visuals",
            command=lambda: self._switch_tab('visuals'),
            width=10
        )
        self.visuals_tab_btn.pack(side=tk.LEFT, padx=2)

        # Compare button
        if self.on_compare:
            ttk.Button(
                tab_frame,
                text="Compare",
                command=self.on_compare,
                width=10
            ).pack(side=tk.LEFT, padx=(20, 2))

        # Spacer
        ttk.Frame(header).grid(row=0, column=2, sticky='ew')

        # Run button (accent style)
        self.run_btn = ttk.Button(
            header,
            text="▶ RUN SIMULATION",
            command=self._on_run,
            style='Accent.TButton'
        )
        self.run_btn.grid(row=0, column=3, sticky='e', padx=5)

        # Progress indicator (hidden initially)
        self.progress_frame = ttk.Frame(header)
        self.progress = ttk.Progressbar(
            self.progress_frame,
            mode='determinate',
            length=150
        )
        self.progress.pack(side=tk.LEFT, padx=5)
        self.progress_label = ttk.Label(self.progress_frame, text="0%")
        self.progress_label.pack(side=tk.LEFT)

    def _create_tabs(self):
        """Create tab content frames."""
        # Container for tab content
        self.tab_container = ttk.Frame(self)
        self.tab_container.grid(row=1, column=0, sticky='nsew', padx=5, pady=5)
        self.tab_container.columnconfigure(0, weight=1)
        self.tab_container.rowconfigure(0, weight=1)

        # Lineup tab content
        self.lineup_frame = ttk.Frame(self.tab_container)
        self._create_lineup_content()

        # Visuals tab content
        self.visuals_frame = ttk.Frame(self.tab_container)
        self._create_visuals_content()

        # Show lineup tab by default
        self._switch_tab('lineup')

    def _create_lineup_content(self):
        """Create lineup tab content."""
        self.lineup_frame.columnconfigure(0, weight=1)
        self.lineup_frame.columnconfigure(1, weight=0)
        self.lineup_frame.rowconfigure(0, weight=1)

        # Lineup builder
        self.lineup_builder = LineupBuilder(self.lineup_frame)
        self.lineup_builder.grid(row=0, column=0, sticky='nsew', padx=(0, 10))

        # Control buttons
        controls = ttk.Frame(self.lineup_frame)
        controls.grid(row=0, column=1, sticky='ns')

        ttk.Button(
            controls,
            text="▲ Move Up",
            command=self.lineup_builder.move_up,
            width=12
        ).pack(pady=2)

        ttk.Button(
            controls,
            text="▼ Move Down",
            command=self.lineup_builder.move_down,
            width=12
        ).pack(pady=2)

        ttk.Button(
            controls,
            text="✖ Remove",
            command=self.lineup_builder.remove_player,
            width=12
        ).pack(pady=2)

        ttk.Separator(controls, orient='horizontal').pack(fill='x', pady=10)

        ttk.Button(
            controls,
            text="Clear All",
            command=self.lineup_builder.clear_lineup,
            width=12
        ).pack(pady=2)

    def _create_visuals_content(self):
        """Create visuals tab content (placeholder)."""
        self.visuals_frame.columnconfigure(0, weight=1)
        self.visuals_frame.rowconfigure(0, weight=1)

        # Placeholder
        placeholder = ttk.Label(
            self.visuals_frame,
            text="Charts and visualizations will appear here after running a simulation",
            style='Muted.TLabel'
        )
        placeholder.grid(row=0, column=0, sticky='nsew', padx=20, pady=20)

        # This will be populated by ResultsPanel content later
        self.visuals_content = None

    def _switch_tab(self, tab_name: str):
        """Switch between tabs.

        Args:
            tab_name: 'lineup' or 'visuals'
        """
        # Hide both
        self.lineup_frame.grid_forget()
        self.visuals_frame.grid_forget()

        # Show selected
        if tab_name == 'lineup':
            self.lineup_frame.grid(row=0, column=0, sticky='nsew')
            self.lineup_tab_btn.state(['pressed'])
            self.visuals_tab_btn.state(['!pressed'])
        else:
            self.visuals_frame.grid(row=0, column=0, sticky='nsew')
            self.lineup_tab_btn.state(['!pressed'])
            self.visuals_tab_btn.state(['pressed'])

    def _on_run(self):
        """Handle Run button click."""
        if self.on_run:
            self.on_run()

    def update_progress(self, current: int, total: int):
        """Update progress indicator.

        Args:
            current: Current iteration
            total: Total iterations
        """
        # Show progress frame if hidden
        if not self.progress_frame.winfo_viewable():
            self.progress_frame.grid(row=0, column=4, sticky='e', padx=5)

        percentage = (current / total) * 100
        self.progress['value'] = percentage
        self.progress_label.config(text=f"{percentage:.0f}%")
        self.update()

    def hide_progress(self):
        """Hide progress indicator."""
        self.progress_frame.grid_forget()
        self.progress['value'] = 0

    def get_lineup_data(self) -> List[Optional[Player]]:
        """Get current lineup."""
        return self.lineup_builder.get_lineup()

    def set_lineup_data(self, lineup: List[Optional[Player]]):
        """Set lineup data."""
        self.lineup_builder.set_lineup(lineup)

    def load_roster(self, roster: List[Player], team_data):
        """Load roster data into lineup builder.

        Args:
            roster: List of Player objects
            team_data: Raw team data DataFrame
        """
        self.lineup_builder.load_data(roster, team_data)
```
